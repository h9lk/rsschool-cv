function comp(array1, array2) {
          if (!Array.isArray(array1) || !Array.isArray(array2)) {
            return false
          }
          const a = array1.map((item) ⇒ Math.pow(item, 2)).sort((a, b) ⇒ b − a);
          const b = array2.sort((a, b) ⇒ b − a);
          return a.every((item, i) ⇒ item ≡ b[i]);
        }
